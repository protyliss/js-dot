Class WeakIdMap<V>

WeakMap with WeakKey as string

Example

Default
const map = new WeakIdMap();

map.set('foo', true);
map.set('bar', false);

// returns true
map.has('foo');

map.delete('bar');

// return false
map.has('bar');

Example

Sharing ID
const ids = new WeakIds();
const a = new WeakIdMap();
const b = new WeakIdMap();

a.set('foo', 1);
b.set('foo', 2);

ids.delete('foo');

// returns false
a.has('foo');
b.has('foo');
